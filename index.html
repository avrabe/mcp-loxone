Installation Commands 
# Clone the repository
git clone https://github.com/avrabe/mcp-loxone-gen1.git
cd mcp-loxone-gen1

# Install dependencies
uv sync

# Configure Loxone credentials
uvx --from . loxone-mcp setup

# Start the MCP server
uv run mcp dev src/loxone_mcp/server.py

Scene Management Example 
# Activate complex house scenes
activate_house_scene("night_mode")    # Lights off, blinds closed, sleep mode on
activate_house_scene("morning_mode")  # Lights on, blinds open, sleep mode off
activate_house_scene("all_off")       # Turn everything off

# Check current scene status
get_scene_status_overview()

# Control specific components
set_lighting_mood("Wohnzimmer", 2)    # Activate preset 2 in living room
control_central_lighting("on")        # Turn on all house lights
